const SHEET_ID = "1fDzL5O-NC7IWhYIQXi4SbQ91lCFxwN9aMPbzY2Y-L0M"; const SHEET_NAME = "Bookings"; const TZ = "America/New_York"; const SLOT_MINUTES = 30; function doGet(e) { try { const action = e.parameter.action; const callback = e.parameter.callback; const data = e.parameter.data ? JSON.parse(e.parameter.data) : {}; let result; if (action === "listBookings") { result = listBookings(); } else if (action === "createBooking") { result = createBooking(data.booking); } else if (action === "cancelBooking") { result = cancelBooking(data.id, data.cancelledBy); } else if (action === "updateBooking") { result = updateBooking(data.booking); } else if (action === "updateBookingStatus") { result = updateBookingStatus(data.id, data.status); } else if (action === "deleteBooking") { result = deleteBooking(data.id); } else { result = { ok: false, error: "Invalid action" }; } return output(result, callback); } catch (err) { return output({ ok: false, error: err.message }, e.parameter.callback); } } function output(obj, callback) { const json = JSON.stringify(obj); const text = callback ? `${callback}(${json});` : json; return ContentService .createTextOutput(text) .setMimeType(ContentService.MimeType.JAVASCRIPT); } function getSheet() { const ss = SpreadsheetApp.openById(SHEET_ID); if (ss.getSpreadsheetTimeZone() !== TZ) { ss.setSpreadsheetTimeZone(TZ); } let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } if (sheet.getLastRow() === 0) { sheet.appendRow([ "ID", "Date", "Hour", "Minute", "Name", "NJIT ID", "Room Number", "Status", "Created At", "Cancelled At", "Cancelled By", "Completed At", "Updated At" ]); } // Force the Date column (B) to plain text so Sheets doesn't auto-convert // dates and shift them by timezone. sheet.getRange("B:B").setNumberFormat("@"); return sheet; } function normalizeDate(value) { if (value instanceof Date) { return Utilities.formatDate(value, TZ, "yyyy-MM-dd"); } const s = String(value || "").trim(); if (/^\d{4}-\d{2}-\d{2}/.test(s)) { return s.substring(0, 10); } const d = new Date(s); if (!isNaN(d.getTime())) { return Utilities.formatDate(d, TZ, "yyyy-MM-dd"); } return s; } function fmtDateTime(value) { if (value instanceof Date) { return Utilities.formatDate(value, TZ, "yyyy-MM-dd HH:mm:ss"); } return String(value || ""); } function pad2(n) { n = Number(n); return (n < 10 ? "0" : "") + n; } // Returns true if any rows were updated. function autoCompletePastBookings(sheet, values) { const now = new Date(); const nowStr = Utilities.formatDate(now, TZ, "yyyy-MM-dd HH:mm:ss"); let changed = false; for (let i = 1; i < values.length; i++) { const status = String(values[i][7] || ""); if (status !== "Booked") continue; const dateStr = normalizeDate(values[i][1]); const hour = Number(values[i][2]); const min = Number(values[i][3]); if (!dateStr || isNaN(hour) || isNaN(min)) continue; let endHour = hour; let endMin = min + SLOT_MINUTES; if (endMin >= 60) { endHour += 1; endMin -= 60; } const endStr = dateStr + " " + pad2(endHour) + ":" + pad2(endMin) + ":00"; if (endStr <= nowStr) { sheet.getRange(i + 1, 8).setValue("Completed"); sheet.getRange(i + 1, 12).setValue(now); sheet.getRange(i + 1, 13).setValue(now); values[i][7] = "Completed"; values[i][11] = now; values[i][12] = now; changed = true; } } return changed; } function normalizeBookingsRows(values) { return values.slice(1).map(function(row) { return { id: String(row[0] || ""), date: normalizeDate(row[1]), hour: Number(row[2]), min: Number(row[3]), name: String(row[4] || ""), njit: String(row[5] || ""), roomNumber: String(row[6] || ""), status: String(row[7] || "Booked"), createdAt: fmtDateTime(row[8]), cancelledAt: fmtDateTime(row[9]), cancelledBy: String(row[10] || ""), completedAt: fmtDateTime(row[11]), updatedAt: fmtDateTime(row[12]) }; }); } function listBookings() { const sheet = getSheet(); let values = sheet.getDataRange().getValues(); if (values.length <= 1) { return { ok: true, bookings: [] }; } const changed = autoCompletePastBookings(sheet, values); if (changed) { SpreadsheetApp.flush(); } return { ok: true, bookings: normalizeBookingsRows(values) }; } function createBooking(booking) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(); const values = sheet.getDataRange().getValues(); const date = normalizeDate(booking.date); const hour = Number(booking.hour); const min = Number(booking.min); if (!date || isNaN(hour) || isNaN(min)) { return { ok: false, error: "Invalid booking data." }; } for (let i = 1; i < values.length; i++) { const rowDate = normalizeDate(values[i][1]); const rowHour = Number(values[i][2]); const rowMin = Number(values[i][3]); const rowStatus = String(values[i][7] || ""); if ( rowDate === date && rowHour === hour && rowMin === min && rowStatus === "Booked" ) { return { ok: false, error: "Sorry, this slot has already been booked." }; } } const now = new Date(); sheet.appendRow([ Utilities.getUuid(), date, hour, min, String(booking.name || ""), String(booking.njit || ""), String(booking.roomNumber || ""), "Booked", now, "", "", "", now ]); SpreadsheetApp.flush(); return listBookings(); } finally { lock.releaseLock(); } } function cancelBooking(id, cancelledBy) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(); const values = sheet.getDataRange().getValues(); const now = new Date(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(id)) { sheet.getRange(i + 1, 8).setValue("Cancelled"); sheet.getRange(i + 1, 10).setValue(now); sheet.getRange(i + 1, 11).setValue(cancelledBy || "user"); sheet.getRange(i + 1, 13).setValue(now); SpreadsheetApp.flush(); return listBookings(); } } return { ok: false, error: "Booking not found." }; } finally { lock.releaseLock(); } } function updateBookingStatus(id, status) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(); const values = sheet.getDataRange().getValues(); const now = new Date(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(id)) { sheet.getRange(i + 1, 8).setValue(status); if (status === "Completed") { sheet.getRange(i + 1, 12).setValue(now); } sheet.getRange(i + 1, 13).setValue(now); SpreadsheetApp.flush(); return listBookings(); } } return { ok: false, error: "Booking not found." }; } finally { lock.releaseLock(); } } function updateBooking(booking) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(); const values = sheet.getDataRange().getValues(); const now = new Date(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(booking.id)) { sheet.getRange(i + 1, 2).setValue(normalizeDate(booking.date)); sheet.getRange(i + 1, 3).setValue(Number(booking.hour)); sheet.getRange(i + 1, 4).setValue(Number(booking.min)); sheet.getRange(i + 1, 5).setValue(String(booking.name || "")); sheet.getRange(i + 1, 6).setValue(String(booking.njit || "")); sheet.getRange(i + 1, 7).setValue(String(booking.roomNumber || "")); sheet.getRange(i + 1, 8).setValue(String(booking.status || "Booked")); sheet.getRange(i + 1, 13).setValue(now); SpreadsheetApp.flush(); return listBookings(); } } return { ok: false, error: "Booking not found." }; } finally { lock.releaseLock(); } } function deleteBooking(id) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(); const values = sheet.getDataRange().getValues(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(id)) { sheet.deleteRow(i + 1); SpreadsheetApp.flush(); return listBookings(); } } return { ok: false, error: "Booking not found." }; } finally { lock.releaseLock(); } }
